Community Leaderboard

Total numbers of package submissions and reviews by community members. By default this includes editors, with a toggle to filter out numbers from anybody who is or has been a member of the editorial team.

viewof showEditors = Inputs.toggle({label: "Include editors", value: true})
leaderboard = {
    const counts = new Map();
    const editors = new Set();

    const add = (handle, role) => {
        if (!handle || handle === "NA") return;
        if (!counts.has(handle)) counts.set(handle, {handle, author: 0, reviewer: 0});
        counts.get(handle)[role]++;
    };

    submissions.forEach(row => {
        add(row.aut_gh,    "author");
        add(row.reviewer1, "reviewer");
        add(row.reviewer2, "reviewer");
        if (row.editor && row.editor !== "NA") editors.add(row.editor);
    });

    return Array.from(counts.values())
        .filter(d => showEditors || !editors.has(d.handle))
        .map(d => ({...d, total: d.author + d.reviewer}))
        .filter(d => d.total > 0)
        .sort((a, b) => b.total - a.total || b.author - a.author)
        .slice(0, 40);
}
leaderboardLong = leaderboard.flatMap(d =>
    ["author", "reviewer"]
        .map(role => ({handle: d.handle, role, count: d[role]}))
        .filter(d => d.count > 0)
)
